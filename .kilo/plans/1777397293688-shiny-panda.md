### 3. PDF Path Inconsistency

**Issue**: PDF path is stored inconsistently - sometimes as filename, sometimes as full path.

**Evidence**:
```typescript

  buffer: Buffer,
  filename: string

  await fs.mkdir(STORAGE_PATH, { recursive: true });
  const filePath = path.join(STORAGE_PATH, filename);  // Full path

// But getPDFPath just joins with STORAGE_PATH again
```

**Problem**: The full path from `savePDFToStorage` is saved to `receipt.pdfPath`, but when it's used for email attachments, it's treated as a relative path.

```typescript
// Add intermediate receipt status
enum ReceiptStatus {
  DRAFT = 'draft',
  ISSUING = 'issuing',  // NEW: Processing issuance
  ISSUED = 'issued',      // Final state
  VOIDED = 'voided'
}

// Update Prisma schema to include 'issuing' status

export async function issueDraftReceipt(
  id: number,
  userId: number,
  input?: IssueReceiptInput
): Promise<ReceiptResponseDTO> {
  const receipt = await prisma.receipt.findUnique({
    where: { id },
  });

  if (!receipt) {
    throw new Error('Receipt not found');
  }

  if (receipt.status !== 'draft') {
    throw new Error('Only draft receipts can be issued');
  }

  // Step 1: Mark as 'issuing' (atomic)
  const updatedReceipt = await prisma.$transaction(async (tx) => {
    const receiptNumber = await generateNextReceiptNumberWithTx(tx);
    const updatedReceipt = await tx.receipt.update({
      where: { id },
      data: {
        receiptNumber,
        status: 'issuing',  // NEW: Intermediate state
        issuedAt: new Date(),
        issuedBy: userId,
        customerId,
        customerSnapshot: customerSnapshot as any,
        notes: input?.notes ?? receipt.notes,
        version: { increment: 1 },
      },
    });
    return updatedReceipt;
  });

  // Step 2: Generate PDF (outside transaction but with compensation)
  try {
    const receiptDTO = toReceiptDTO(updatedReceipt);
    const templateData = prepareReceiptTemplateData(receiptDTO);
    const pdfResult = await renderReceiptPDF(templateData);
    const savedPath = await savePDFToStorage(pdfResult.buffer, pdfResult.filename);

    const finalReceipt = await prisma.receipt.update({
      where: { id },
      data: {
        pdfPath: pdfResult.filename,
        pdfGeneratedAt: pdfResult.generatedAt,
        status: 'issued',  // Final state
        version: { increment: 1 },
      },
    });

    triggerAutoEmail(id, customerSnapshot as any, userId, 'issued receipt');

    return toReceiptDTO(finalReceipt);
  } catch (pdfError) {
    console.error('Failed to generate PDF on receipt issue:', pdfError);

    // Compensation: Roll back to draft status
    try {
      await prisma.receipt.update({
        where: { id },
        data: {
          status: 'draft',  // Roll back to draft
          issuedAt: null,
          issuedBy: null,
          receiptNumber: `DRAFT-${randomUUID()}`,  // Reset to draft number
        },
      });
    } catch (rollbackError) {
      console.error('Failed to roll back receipt after PDF generation failure:', rollbackError);
      // Alert operations team - receipt is in 'issuing' state but PDF generation failed
    }

    throw new Error(`Receipt issuance failed: PDF generation error - ${pdfError.message}`);
  }
}
```

```typescript
import fs from 'fs/promises';
import path from 'path';

const STORAGE_PATH = process.env.PDF_STORAGE_PATH || '/app/storage/receipts';

async function processJob(
  job: Awaited<ReturnType<typeof prisma.emailQueue.findUnique>> & { id: string },
): Promise<void> {
  try {
    await prisma.emailQueue.update({
      where: { id: job.id },
      data: { status: 'processing' },
    });

    const options: Parameters<typeof sendEmail>[0] = {
      to: job.recipientEmail,
      subject: job.subject,
      html: job.htmlContent,
      text: job.textContent,
    };

    // Handle attachments with path resolution and existence check
    if (job.attachmentPath && job.attachmentFilename) {
      const fullAttachmentPath = path.join(STORAGE_PATH, job.attachmentPath);

      // Check if file exists
      try {
        await fs.access(fullAttachmentPath, fs.constants.R_OK);
      } catch (error) {
        logError(`PDF attachment not found: ${fullAttachmentPath}`, {
          jobId: job.id,
          receiptId: job.receiptId,
        });

        // Mark job as failed
        await prisma.$transaction(async (tx) => {
          await tx.emailQueue.update({
            where: { id: job.id },
            data: {
              status: 'failed',
              lastError: `PDF file not found: ${fullAttachmentPath}`,
              processedAt: new Date(),
            },
          });

            where: { id: job.receiptId },
            data: {
              emailStatus: 'failed',
              emailErrorMessage: `PDF file not found`,
              emailAttempts: { increment: 1 },
            },

      }

      options.attachments = [
        {
          filename: job.attachmentFilename,
          path: fullAttachmentPath,
        },
      ];
    }

    const result = await sendEmail(options);

    if (result.success) {

    } else {
      // ... existing error handling ...
    }
  } catch (error) {
    // ... existing error handling ...

}
```

```typescript
    },
    select: {
      id: true,
      receiptNumber: true,
      pdfPath: true,
      issuedAt: true,
    },
    orderBy: { issuedAt: 'desc' },
    take: limit,
  });

  let recovered = 0;
  let failed = 0;
  let skipped = 0;

  for (const receipt of receiptsWithPDF) {
    const fullPath = path.join(STORAGE_PATH, receipt.pdfPath);

    try {
      // Check if file exists
      await fs.access(fullPath, fs.constants.R_OK);

      // File exists, skip
      skipped++;
      continue;
    } catch {
      // File doesn't exist, need to regenerate
      logInfo(`PDF file missing for receipt ${receipt.receiptNumber}, regenerating...`, {
        receiptId: receipt.id,
        receiptNumber: receipt.receiptNumber,
        missingPath: fullPath,
      });

      try {
        // Get full receipt data
        const fullReceipt = await prisma.receipt.findUnique({
          where: { id: receipt.id },
          include: {
            businessSnapshot: true,
            customerSnapshot: true,
          },
        });

        if (!fullReceipt) {
          logError(`Receipt not found during recovery: ${receipt.id}`);
          failed++;
          continue;
        }

        // Import PDF generation function
        const { renderReceiptPDF } = await import('./receiptTemplateService');
        const { savePDFToStorage } = await import('./pdfService');
        const { prepareReceiptTemplateData } = await import('./types/receipt-template');
        const { toReceiptDTO } = await import('./types/receipt');

        // Prepare template data
        const receiptDTO = toReceiptDTO(fullReceipt);
        const templateData = prepareReceiptTemplateData(receiptDTO);

        // Regenerate PDF
        const pdfResult = await renderReceiptPDF(templateData);

        // Save new PDF
        await savePDFToStorage(pdfResult.buffer, pdfResult.filename);

        // Update receipt with new PDF generation timestamp
        await prisma.receipt.update({
          where: { id: receipt.id },
          data: {
            pdfGeneratedAt: pdfResult.generatedAt,
            version: { increment: 1 },
          },
        });

        recovered++;
        logInfo(`Successfully recovered PDF for receipt ${receipt.receiptNumber}`, {
          receiptId: receipt.id,
          receiptNumber: receipt.receiptNumber,
        });
      } catch (recoveryError) {
        logError(`Failed to recover PDF for receipt ${receipt.receiptNumber}`, {
          receiptId: receipt.id,
          receiptNumber: receipt.receiptNumber,
          error: recoveryError instanceof Error ? recoveryError.message : 'Unknown error',
        });
        failed++;
      }
    }
  }

  const summary = { recovered, failed, skipped };
  logInfo('PDF recovery completed', summary);

  return summary;
}

/**
 * One-time recovery command for admin use
 */
export async function runOneTimeRecovery(): Promise<void> {
  logInfo('Running one-time PDF recovery...');

  // Recover in batches to avoid memory issues
  let totalRecovered = 0;
  let totalFailed = 0;
  let totalSkipped = 0;
  let hasMore = true;
  let offset = 0;

  while (hasMore) {
    const result = await recoverMissingPDFs(100);

    totalRecovered += result.recovered;
    totalFailed += result.failed;
    totalSkipped += result.skipped;

    hasMore = result.recovered + result.failed > 0;
    offset += 100;

    if (result.recovered + result.failed === 0) {
      break;
    }

    // Small delay between batches
    await new Promise(resolve => setTimeout(resolve, 1000));
  }

  logInfo('One-time PDF recovery completed', {
    totalRecovered,
    totalFailed,
    totalSkipped,
  });
}
```

```typescript
router.get('/health/pdfs', async (req: Request, res: Response) => {
  try {
    const fs = await import('fs/promises');
    const path = await import('path');

    const STORAGE_PATH = process.env.PDF_STORAGE_PATH || '/app/storage/receipts';

    // Count receipts with PDF path
    const receiptsWithPDF = await prisma.receipt.count({
      where: {
        status: 'issued',
        pdfPath: { not: null },
      },
    });

    // Count actual PDF files
    const pdfFiles = await fs.readdir(STORAGE_PATH);
    const pdfFileCount = pdfFiles.filter(f => f.endsWith('.pdf')).length;

    // Check for missing files
    const recentReceipts = await prisma.receipt.findMany({
      where: {
        status: 'issued',
        pdfPath: { not: null },
        issuedAt: { gte: new Date(Date.now() - 7 * 24 * 60 * 60 * 1000) }, // Last 7 days
      },
      select: {
        id: true,
        receiptNumber: true,
        pdfPath: true,
        issuedAt: true,
      },
      take: 100,
      orderBy: { issuedAt: 'desc' },
    });

    let missingPDFCount = 0;
    for (const receipt of recentReceipts) {
      const fullPath = path.join(STORAGE_PATH, receipt.pdfPath);
      try {
        await fs.access(fullPath, fs.constants.R_OK);
      } catch {
        missingPDFCount++;
      }
    }

    const health = {
      status: 'OK',
      receiptsWithPDF,
      pdfFileCount,
      missingPDFCount,
      missingPDFPercentage: receiptsWithPDF > 0 ? (missingPDFCount / receiptsWithPDF * 100).toFixed(2) : 0,
      timestamp: new Date().toISOString(),
    };

    // Log warning if too many missing files
    if (missingPDFCount > 10) {
      logError('PDF health check: Too many missing PDF files', health);
    } else if (missingPDFCount > 0) {
      logInfo('PDF health check: Some PDF files missing', health);
    }

    res.json(health);
  } catch (error) {
    logError('PDF health check failed', error);
    res.status(500).json({ error: 'Health check failed' });
  }
});
```
